docker images # списов скаченных образов
docker ps # список всех запущенных контейнеров
docker ps -a # список вообще всех контейнеров, в том числе и остановленных
docker run -it debian # создание и запуск контейнера с debian в интерактивном режиме
docker exec -it <container_id> bash # Запуск bash в уже запущенном контейнере. Exec выполняет команду внутри уже запущенного контейнера
